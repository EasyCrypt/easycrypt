\begin{tactic}{bypr}
  \begin{tsyntax}{bypr}
  Derives a program judgment from a probability relation or an exact
  probability. Only applies to judgments on procedures.

  \textbf{Examples:}
  \begin{mathpar}
    \inferrule%%
      {\forall \mem{m_1}, \mem{m_2}, a.\,
          r_1 = a \Rightarrow
          r_2 = a \Rightarrow
          \Rel{Q}{\mem{m_1}}{\mem{m_2}} \\%
       \forall \vec{a}_1, \vec{a}_2, \mem{m_1}, \mem{m_2}, a.\,
          \Rel{P}{\mem{m_1}[\Arg\mapsto\vec{a}_1]}{\mem{m_2}[\Arg\mapsto\vec{a}_2]} \Rightarrow%
         \PR{f_1}{\vec{a}_1}{\mem{m_1}}{a = r_1} = \PR{f_2}{\vec{a}_2}{\mem{m_2}}{a = r_2}} %%
      {\pRHL{P}{f_1}{f_2}{Q}}%%
      \quad\mbox{(\prhl)\quad\parbox{200pt}{\tct{bypr (r$_1$) (r$_2$)}}} \\
    \inferrule%%
      {\forall \mem{m}, \vec{a}.\,\Pred{P}{m[\Arg\mapsto\vec{a}]} \Rightarrow
          \PR{f}{\vec{a}}{m}{E} \mathrel{\diamond} \delta}%%
      {\pHL{P}{f}{E}{\diamond}{\delta}}
      \quad\mbox{(\phl)\quad\parbox{200pt}{\tct{bypr}}} \\
    \inferrule%%
      {\forall \mem{m}, \vec{a}.\,\Pred{P}{m[\Arg\mapsto\vec{a}]} \Rightarrow
          \PR{f}{\vec{a}}{m}{\neg E} \mathop{=}0$\%$r}%%
      {\HL{P}{f}{E}}
      \quad\mbox{(\hl)\quad\parbox{200pt}{\tct{bypr}}} \\
  \end{mathpar}
  \end{tsyntax}
\end{tactic}
